USAGE EXAMPLES

Example 1: Find Bottleneck Activities
────────────────────────────────────
let transitions = calculate_activity_transition_times(&log);
for trans in transitions.iter().take(5) {
    println!("{} -> {}: {:.1}s avg",
        trans.from_activity, trans.to_activity, trans.avg_time_seconds);
}

Example 2: Check SLA Compliance
───────────────────────────────
let compliance = check_sla_compliance(&log, Some(86400.0));  // 24h SLA
let rate = (compliance.iter().filter(|c| c.is_compliant).count() as f64
    / compliance.len() as f64) * 100.0;
println!("SLA compliance: {:.1}%", rate);

Example 3: Find Complex Cases
─────────────────────────────
let complexity = calculate_case_complexity(&log);
for case in complexity.iter().take(10) {
    println!("Case {}: {:.1} complexity",
        case.case_id, case.complexity_score);
}

Example 4: Detect Anomalies
────────────────────────────
let outliers = detect_outlier_cases_by_duration(&log, Some(2.0));
println!("Found {} outlier cases", outliers.len());

Example 5: Analyze Peak Hours
──────────────────────────────
let peaks = get_peak_hours_analysis(&log);
println!("Busiest hour: {:02}:00 ({} events)",
    peaks[0].hour_of_day, peaks[0].event_count);
